class ComplianceFramework(str, Enum):
    SOC2 = "soc2"
    ISO27001 = "iso27001"
    GDPR = "gdpr"
    HIPAA = "hipaa"
    PCI_DSS = "pci_dss"
    NIST_CSF = "nist_csf"


class ControlStatus(str, Enum):
    COMPLIANT = "compliant"
    NON_COMPLIANT = "non_compliant"
    PARTIAL = "partial"


class Severity(str, Enum):
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"


@dataclass(frozen=True)
class PolicyRule:
    """Immutable policy rule definition."""
    rule_id: str
    framework: ComplianceFramework
    control_id: str
    description: str
    rego_query: str
    severity: Severity
    auto_remediate: bool = False
    created_at: datetime = field(default_factory=datetime.utcnow)


class ComplianceControl(BaseModel):
    """Compliance control model with validation."""
    control_id: str = Field(..., min_length=3, max_length=50)
    framework: ComplianceFramework
    title: str = Field(..., min_length=10, max_length=200)
    description: str = Field(..., min_length=50)
    requirements: List[str] = Field(default_factory=list, min_items=1)
    owner: str
    review_frequency_days: int = Field(default=90, ge=30, le=365)


class EvidenceArtifact(BaseModel):
    """Evidence artifact for audit trails."""
    artifact_id: UUID = Field(default_factory=uuid4)
    control_id: str
    framework: ComplianceFramework
    evidence_type: str
    storage_path: str
    checksum: str
    collected_at: datetime = Field(default_factory=datetime.utcnow)
    collector_agent: str
    metadata: Dict[str, Any] = Field(default_factory=dict)
    
    def verify_integrity(self) -> bool:
        """Verify SHA-256 checksum of evidence."""
        try:
            with open(self.storage_path, 'rb') as f:
                calculated = hashlib.sha256(f.read()).hexdigest()
            return calculated == self.checksum
        except Exception as e:
            logger.error(f"Integrity check failed: {e}")
            return False


class ComplianceFinding(BaseModel):
    """Compliance finding with remediation tracking."""
    finding_id: UUID = Field(default_factory=uuid4)
    rule_id: str
    control_id: str
    framework: ComplianceFramework
    severity: Severity
    status: ControlStatus
    description: str
    resource_id: Optional[str] = None
    remediation_steps: List[str] = Field(default_factory=list)
    assigned_to: Optional[str] = None
    due_date: Optional[datetime] = None
    created_at: datetime = Field(default_factory=datetime.utcnow)
    resolved_at: Optional[datetime] = None


class PolicyEngine:
    """
    Core policy enforcement engine with OPA integration.
    Development Value: $15,000
    """
    
    def __init__(
        self,
        opa_url: str = "http://localhost:8181",
        db_dsn: str = "postgresql://localhost/cap1",
        redis_url: str = "redis://localhost:6379"
    ):
        self.opa_client = OPAClient(opa_url)
        self.db_dsn = db_dsn
        self.redis_url = redis_url
        self.db_pool: Optional[asyncpg.Pool] = None
        self.redis: Optional[aioredis.Redis] = None
        self._rules: Dict[str, PolicyRule] = {}
        
    async def initialize(self) -> None:
        """Initialize database connections and load policies."""
        self.db_pool = await asyncpg.create_pool(
            self.db_dsn, min_size=5, max_size=20, command_timeout=60
        )
        self.redis = await aioredis.from_url(
            self.redis_url, encoding="json", decode_responses=True
        )
        await self._load_policies_from_db()
        logger.info("PolicyEngine initialized successfully")
    
    async def _load_policies_from_db(self) -> None:
        """Load active policies from database."""
        async with self.db_pool.acquire() as conn:
            rows = await conn.fetch(
                "SELECT * FROM policy_rules WHERE active = TRUE"
            )
            for row in rows:
                rule = PolicyRule(
                    rule_id=row['rule_id'],
                    framework=ComplianceFramework(row['framework']),
                    control_id=row['control_id'],
                    description=row['description'],
                    rego_query=row['rego_query'],
                    severity=Severity(row['severity']),
                    auto_remediate=row['auto_remediate']
                )
                self._rules[rule.rule_id] = rule
        logger.info(f"Loaded {len(self._rules)} active policies")
    
    async def evaluate_compliance(
        self,
        resource_type: str,
        resource_data: Dict[str, Any],
        frameworks: Optional[Set[ComplianceFramework]] = None
    ) -> List[ComplianceFinding]:
        """
        Evaluate resource compliance against all applicable policies.
        
        Args:
            resource_type: Type of resource (e.g., 's3_bucket', 'ec2_instance')
            resource_data: Resource configuration data
            frameworks: Specific frameworks to check (default: all)
            
        Returns:
            List of compliance findings
        """
        findings = []
        frameworks = frameworks or set(ComplianceFramework)
        
        # Check cache first
        cache_key = f"compliance:{resource_type}:{hash(str(resource_data))}"
        cached = await self.redis.get(cache_key)
        if cached:
            return [ComplianceFinding.parse_raw(f) for f in json.loads(cached)]
        
        # Evaluate each applicable rule
        for rule in self._rules.values():
            if rule.framework not in frameworks:
                continue
                
            try:
                # Query OPA for policy decision
                result = await self.opa_client.query(
                    rule.rego_query,
                    {"resource": resource_data}
                )
                
                if not result.get("compliant", False):
                    finding = ComplianceFinding(
                        rule_id=rule.rule_id,
                        control_id=rule.control_id,
                        framework=rule.framework,
                        severity=rule.severity,
                        status=ControlStatus.NON_COMPLIANT,
                        description=result.get("message", rule.description),
                        resource_id=resource_data.get("id"),
                        remediation_steps=result.get("remediation", [])
                    )
                    findings.append(finding)
                    
                    # Auto-remediate if enabled and severity is critical/high
                    if rule.auto_remediate and rule.severity in (Severity.CRITICAL, Severity.HIGH):
                        await self._trigger_auto_remediation(finding)
                        
            except Exception as e:
                logger.error(f"Policy evaluation failed for {rule.rule_id}: {e}")
                findings.append(ComplianceFinding(
                    rule_id=rule.rule_id,
                    control_id=rule.control_id,
                    framework=rule.framework,
                    severity=Severity.HIGH,
                    status=ControlStatus.NON_COMPLIANT,
                    description=f"Policy evaluation error: {str(e)}"
                ))
        
        # Cache results for 5 minutes
        await self.redis.setex(
            cache_key, 300, json.dumps([f.json() for f in findings])
        )
        
        return findings
    
    async def _trigger_auto_remediation(self, finding: ComplianceFinding) -> None:

    async def generate_evidence_package(
        self,
        framework: ComplianceFramework,
        start_date: datetime,
        end_date: datetime
    ) -> Path:
        """Generate audit-ready evidence package."""
        async with self.db_pool.acquire() as conn:
            artifacts = await conn.fetch(
                """
                SELECT * FROM evidence_artifacts 
                WHERE framework = $1 
                AND collected_at BETWEEN $2 AND $3
                ORDER BY control_id, collected_at
                """,
                framework.value, start_date, end_date
            )
        
        # Generate tamper-evident package
        package_id = uuid4()
        package_path = Path(f"/tmp/evidence_packages/{package_id}")
        package_path.mkdir(parents=True, exist_ok=True)
        
        manifest = {
            "package_id": str(package_id),
            "framework": framework.value,
            "generated_at": datetime.utcnow().isoformat(),
            "artifacts": []
        }
        
        for artifact_record in artifacts:
            artifact = EvidenceArtifact.parse_raw(json.dumps(dict(artifact_record)))
            
            # Verify integrity before inclusion
            if artifact.verify_integrity():
                manifest["artifacts"].append({
                    "artifact_id": str(artifact.artifact_id),
                    "control_id": artifact.control_id,
                    "checksum": artifact.checksum,
                    "collected_at": artifact.collected_at.isoformat()
                })
        
        # Write manifest with signature
        manifest_path = package_path / "manifest.json"
        manifest_path.write_text(json.dumps(manifest, indent=2))
        
        # Create package checksum
        package_hash = hashlib.sha256(manifest_path.read_bytes()).hexdigest()
        (package_path / "package.sig").write_text(package_hash)
        
        logger.info(f"Generated evidence package: {package_id}")
        return package_path

async def verify_token(credentials: HTTPAuthorizationCredentials = Depends(security)) -> dict:
    """Verify JWT token and return claims."""
    try:
        payload = jwt.decode(credentials.credentials, "SECRET_KEY", algorithms=["HS256"])
        return payload
    except jwt.InvalidTokenError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid authentication credentials"
        )

@app.middleware("http")
async def metrics_middleware(request, call_next):
    """Collect request metrics."""
    start_time = time.time()
    response = await call_next(request)
    duration = time.time() - start_time
    
    REQUEST_COUNT.labels(
        method=request.method,
        endpoint=request.url.path,
        status=response.status_code
    ).inc()
    REQUEST_DURATION.observe(duration)
    
    return response

@app.post("/v1/compliance/evaluate")
async def evaluate_resource(
    resource_type: str,
    resource_data: dict,
    frameworks: Optional[List[ComplianceFramework]] = None,
    user: dict = Depends(verify_token)
):
    """Evaluate a resource against compliance policies."""
    engine: PolicyEngine = app.state.engine
    framework_set = set(frameworks) if frameworks else None
    findings = await engine.evaluate_compliance(resource_type, resource_data, framework_set)
    return findings

@app.get("/v1/compliance/score/{framework}")
async def get_compliance_score(framework: ComplianceFramework, user: dict = Depends(verify_token)):
    """Get overall compliance score for a framework."""
    engine: PolicyEngine = app.state.engine
    async with engine.db_pool.acquire() as conn:
        row = await conn.fetchrow(
            """
            SELECT 
                COUNT(*) as total_controls,
                SUM(CASE WHEN status = 'compliant' THEN 1 ELSE 0 END) as compliant
            FROM control_status
            WHERE framework = $1
            """,
            framework.value
        )
    
    if row and row['total_controls'] > 0:
        score = (row['compliant'] / row['total_controls']) * 100
        COMPLIANCE_SCORE.labels(framework=framework.value).set(score)
        return {
            "framework": framework.value,
            "score": round(score, 2),
            "total_controls": row['total_controls'],
            "compliant_controls": row['compliant']
        }
    
    return {"framework": framework.value, "score": 0, "total_controls": 0}

class RegulatoryScoutAgent:
    """
    Agent that continuously monitors regulatory sources for changes.
    Development Value: $8,000
    """
    
    MONITORED_SOURCES = [
        "https://www.federalregister.gov/documents/full_text/xml",
        "https://www.sec.gov/cgi-bin/browse-edgar?action=getcurrent",
        "https://www.fda.gov/medical-devices/news-events-medical-devices/medical-device-safety/rss.xml"
    ]
    
    def __init__(self, openai_api_key: str):
        self.openai_client = openai.AsyncOpenAI(api_key=openai_api_key)
        self.known_documents: set = set()
        self.impact_threshold = 0.7
        
    async def monitor_loop(self):
        """Continuous monitoring loop."""
        while True:
            try:
                await self.check_all_sources()
                await asyncio.sleep(3600)  # Check every hour
            except Exception as e:
                logger.error(f"Monitoring error: {e}")
                await asyncio.sleep(300)
    
    async def check_all_sources(self):
        """Check all regulatory sources for updates."""
        tasks = [self.check_source(url) for url in self.MONITORED_SOURCES]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        for url, result in zip(self.MONITORED_SOURCES, results):
            if isinstance(result, Exception):
                logger.error(f"Failed to check {url}: {result}")
            else:
                logger.info(f"Checked {url}: {len(result)} new documents")
    
    async def check_source(self, url: str) -> List[Dict]:
        """Check a single source for new documents."""
        new_documents = []
        
        async with aiohttp.ClientSession() as session:
            async with session.get(url, timeout=30) as response:
                content = await response.text()
        
        documents = self._parse_documents(content, url)
        
        for doc in documents:
            doc_hash = hashlib.sha256(f"{doc['title']}{doc['date']}".encode()).hexdigest()
            
            if doc_hash not in self.known_documents:
                self.known_documents.add(doc_hash)
                impact_score = await self._analyze_impact(doc)
                
                if impact_score >= self.impact_threshold:
                    doc['impact_score'] = impact_score
                    new_documents.append(doc)
        
        return new_documents
    
    def _parse_documents(self, content: str, url: str) -> List[Dict]:
        """Parse documents from feed content."""
        # Implementation would parse RSS/XML feeds
        return []
    
    async def _analyze_impact(self, document: Dict) -> float:
        """Use AI to analyze regulatory impact."""
        prompt = f"""
        Analyze the following regulatory document for impact on fintech companies:
        
        Title: {document['title']}
        Summary: {document.get('summary', 'N/A')}
        
        Rate the impact on a scale of 0.0 to 1.0 where:
        - 0.0: No impact
        - 0.3: Minor documentation changes
        - 0.5: Process modifications required
        - 0.7: Significant operational changes
        - 1.0: Major business model impact
        
        Return only the numerical score.
        """
        
        try:
            response = await self.openai_client.chat.completions.create(
                model="gpt-4",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.0,
                max_tokens=10
            )
            score = float(response.choices[0].message.content.strip())
            return min(max(score, 0.0), 1.0)
        except Exception as e:
            logger.error(f"Impact analysis failed: {e}")
            return 0.5
